\newglossaryentry{executable}{
    name=executable,
    description={set of computer instructions that has been created (usually by a compiler) so that it can be run by a CPU without additional processing}
}

\newglossaryentry{input stream}{
    name={input stream},
    description={way of communication between an executable and its environment, through which it reads data to process}
}

\newglossaryentry{weakness}{
    name=weakness,
    description={condition in an executable that, under certain circumstances, could contribute to the introduction of vulnerabilities}
}

\newglossaryentry{vulnerability}{
    name=vulnerability,
    description={weakness in an executable that could be exploited or triggered by an attacker}
}

\newglossaryentry{attack surface}{
    name={attack surface},
    description={set of points on the boundary of an executable where an attacker can try to attack}
}

\newglossaryentry{fuzzing}{
    name=fuzzing,
    description={automated testing technique that involves providing invalid, unexpected, or random data as inputs to an executable}
}

\newglossaryentry{emulator}{
    name={emulator},
    description={software that permits executables written for one instruction set architecture to be run on another architecture}
}

\newglossaryentry{virtual machine}{
    name={virtual machine},
    description={software that allows a single host to run one or more guest operating systems}
}

\newglossaryentry{instrumentation}{
    name={instrumentation},
    description={insertion new code at any point in an existing executable to observe or modify its behavior}
}

\newglossaryentry{symbolic execution}{
    name={symbolic execution},
    description={technique for executing an executable by finding all execution paths and generate corresponding input for each path}
}

\newglossaryentry{exploitation}{
    name={exploitation},
    description={act of an attacker for maliciously using an executable in a way in which it was not programmed}
}

\newglossaryentry{signature}{
    name={signature},
    description={distinguishing pattern associated with an attack}
}
